```C++
// 计算数字连续出现几次

#include <iostream>

int main()

{

    int curval = 0, val = 0;

    if (std::cin >> curval)  // 选中第一个数

    {

        int count = 1;  // 计数一次

        while (std::cin >> val) // 继续读入下一个数字

        {

            if (val == curval)

            {

                ++count;  //数字一样 计数+1

            }

            else

            {

                std::cout << curval << "出现了" << count << "次" << std::endl;

                curval = val;  // 看下个数字连续几次

                count = 1;  // 计数归1

            }

        }

        std::cout << curval << "出现了" << count << "次" << std::endl;

    }

  

    return 0;

}
```
